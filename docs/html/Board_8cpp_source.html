133bool Board::ExecuteMove(Move &move, Player* executor)
134{
135 int x0 = move.getStartX(), y0 = move.getStartY();
136 int x1 = move.getEndX(), y1 = move.getEndY();
137
138 Piece* moving = m_board[y0][x0];
139 Piece* target = m_board[y1][x1]; // Target piece (if any)
140 Piece* replacement = nullptr;
141
142 if (!moving || moving->Color() != executor->Color()) {
143 return false;
144 }
145
146 // Ensure the move is valid for the selected piece.
147 if (!moving->IsValidMove(target, this, move, replacement)) {
148 return false;
149 }
150
151 bool wasInCheck = IsKingChecked(executor->Color());
152
153 // Execute the move and capture the move state.
154 MoveRecord record = CommitMoveRecord(move, replacement);
155
156 // If the king was in check and the move doesn't resolve it, revert.
157 if (wasInCheck && IsKingChecked(executor->Color())) {
158 RevertMoveRecord(record);
159
160 return false;
161 }
162
163 // Check if the move puts the player's own king in check.
164 if (IsKingChecked(executor->Color())) {
165 // Revert the move using the saved state.
166 RevertMoveRecord(record);
167 return false;
168 }
169
170 // If a capture was made, record it.
171 if (record.capturedPiece)
172 executor->AddCapture(record.capturedPiece);
173
174
175 // Update check state for opponent.
176 SetKingChecked(moving->Color() == PieceColor::WHITE ? PieceColor::BLACK : PieceColor::WHITE);
177
178 // Handle en passant state if applicable.
179 if (dynamic_cast<Pawn*>(moving) && std::abs(y1 - y0) == 2) {
180 m_enPassantAvailable = true;
181 m_enPassantX = x0;
182 m_enPassantY = y0 + (y1 - y0) / 2;
183 } else {
184 ClearEnPassantTarget();
185 }
186
187 return true;
188}

208// Returns true if the king of the specified color is in check.
209bool Board::IsKingChecked(Color color) const
210{
211
212 // Find the king of the specified color.
213 int kingX = -1, kingY = -1;
214 for (int i = 0; i < 8; ++i) {
215 for (int j = 0; j < 8; ++j) {
216 Piece* piece = m_board[i][j];
217 if (piece != nullptr && piece->Color() == color && piece->IsKing()) {
218 kingX = j;
219 kingY = i;
220 break;
221 }
222 }
223 if (kingX != -1)
224 break;
225 }
226
227 // If king was not found (should not happen in a standard game), return false.
228 if (kingX == -1)
229 return false;
230
231 // Check every enemy piece to see if it has a valid move to the king's position.
232 for (int i = 0; i < 8; ++i) {
233 for (int j = 0; j < 8; ++j) {
234 Piece* piece = m_board[i][j];
235 if (piece != nullptr && piece->Color() != color) {
236 // Create a temporary move from the enemy piece's position to the king's position.
237 Move testMove(j, i, kingX, kingY);
238 Piece* replacement = nullptr;
239 if (piece->IsValidMove(m_board[kingY][kingX], const_cast<Board*>(this), testMove, replacement))
240 {
241 return true; // King is under attack.
242 }
243 }
244 }
245 }
246 return false; // No enemy piece can legally move to the king's square.
247}

262MoveRecord Board::CommitMoveRecord(Move &move, Piece* replacement)
263{
264 int x0 = move.getStartX();
265 int y0 = move.getStartY();
266 int x1 = move.getEndX();
267 int y1 = move.getEndY();
268
269 MoveRecord record;
270 record.x0 = x0;
271 record.y0 = y0;
272 record.x1 = x1;
273 record.y1 = y1;
274 record.prevEnPassantAvailable = m_enPassantAvailable;
275 record.prevEnPassantX = m_enPassantX;
276 record.prevEnPassantY = m_enPassantY;
277
278 record.movedPiece = m_board[y0][x0];
279 record.replacementPiece = replacement; // Will be non-null for a promotion.
280 record.enPassantUsed = false;
281
282 // By default, assume the captured piece is on the target square.
283 record.capturedPiece = m_board[y1][x1];
284
285 // Check for en passant capture: if a pawn moves diagonally into an empty square
286 // that is flagged as an en passant target, then the captured pawn is adjacent.
287 if (dynamic_cast<Pawn*>(record.movedPiece) &&
288 std::abs(x1 - x0) == 1 &&
289 record.capturedPiece == nullptr &&
290 IsEnPassantTarget(x1, y1))
291 {
292 // For en passant, the captured pawn is at (x1, y0).
293 record.capturedPiece = m_board[y0][x1];
294 m_board[y0][x1] = nullptr;
295 record.enPassantUsed = true;
296 ClearEnPassantTarget();
297 }
298
299 // Perform the move.
300 // If a promotion occurred (replacement non-null), place the replacement piece.
301 m_board[y1][x1] = (replacement != nullptr) ? replacement : record.movedPiece;
302 m_board[y0][x0] = nullptr;
303
304 return record;
305}

309void Board::RevertMoveRecord(const MoveRecord &record)
310{
311 // If a promotion occurred, the piece on the destination is the replacement.
312 // We delete it and restore the original pawn.
313 if (record.replacementPiece != nullptr) {
314 delete m_board[record.y1][record.x1];
315 m_board[record.y1][record.x1] = record.movedPiece;
316 } else {
317 // For an en passant capture, the destination square should be empty,
318 // and the captured pawn must be restored to its original square.
319 if (record.enPassantUsed) {
320 m_board[record.y0][record.x0] = record.movedPiece;
321 m_board[record.y1][record.x1] = nullptr;
322 m_board[record.y0][record.x1] = record.capturedPiece;
323 } else {
324 // Standard move: restore the moved piece and captured piece.
325 m_board[record.y0][record.x0] = record.movedPiece;
326 m_board[record.y1][record.x1] = record.capturedPiece;
327 }
328 }
329 // Restore the en passant state.
330 m_enPassantAvailable = record.prevEnPassantAvailable;
331 m_enPassantX = record.prevEnPassantX;
332 m_enPassantY = record.prevEnPassantY;
333}

80 // Helper function to calculate the visual length of a string,
81 // ignoring ANSI escape sequences.
82 static inline size_t DisplayLength(const std::string &text)
83 {
84 size_t length = 0;
85 bool in_escape = false;
86 for (size_t i = 0; i < text.size(); ++i)
87 {
88 if (!in_escape)
89 {
90 if (text[i] == '\033')
91 in_escape = true;
92 else
93 ++length;
94 }
95 else
96 {
97 // Typically ANSI codes end with 'm'
98 if (text[i] == 'm')
99 in_escape = false;
100 }
101 }
102 return length;
103 }

144 {
